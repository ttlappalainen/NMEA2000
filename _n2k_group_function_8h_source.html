240 template <typename T> void MatchRequestField(T FieldVal, T MatchVal, T Mask, bool &Match, tN2kGroupFunctionParameterErrorCode &ErrorCode)
241 {
242 if ( (FieldVal&Mask)!=MatchVal ) {
243 ErrorCode=N2kgfpec_RequestOrCommandParameterOutOfRange;
244 Match=false;
245 } else ErrorCode=N2kgfpec_Acknowledge;
246 }

259 {
260 Match&=(strcmp(FieldVal,MatchVal)==0);
261 ErrorCode = ( Match ? N2kgfpec_Acknowledge : N2kgfpec_RequestOrCommandParameterOutOfRange );
