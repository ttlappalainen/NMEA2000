47//*****************************************************************************
48// On Arduino round() is a macro, hence the definition check. On other systems
49// it is a function that may or may not be implemented so we do it ourselves.
50#if !defined(round)
51double round(double val) {
52 return val >= 0
53 ? floor(val + 0.5)
54 : ceil(val - 0.5);
55}

341//*****************************************************************************
342bool tN2kMsg::GetStr(char *StrBuf, size_t Length, int &Index) const {
343 unsigned char vb;
344 bool nullReached = false;
345 StrBuf[0] = '\0';
346 if ((size_t)Index+Length<=(size_t)DataLen) {
347 for (size_t i=0; i<Length; i++) {
348 vb = GetByte(Index);
349 if (! nullReached) {
350 if (vb == 0x00 || vb == '@') {
351 nullReached = true; // either null or '@' (AIS null character)
352 StrBuf[i] = '\0';
353 StrBuf[i+1] = '\0';
354 } else {
355 StrBuf[i] = vb;
356 StrBuf[i+1] = '\0';
357 }
358 } else {
359 StrBuf[i] = '\0';
360 StrBuf[i+1] = '\0';
361 }
362 }
363 return true;
364 } else return false;
365}
366
367//*****************************************************************************
368bool tN2kMsg::GetStr(size_t StrBufSize, char *StrBuf, size_t Length, unsigned char nulChar, int &Index) const {
369 unsigned char vb;
370 bool nullReached = false;
371 if ( StrBufSize==0 || StrBuf==0 ) {
372 Index+=Length;
373 return true;
374 }
375 StrBuf[0] = '\0';
376 if ((size_t)Index+Length<=(size_t)DataLen) {
377 size_t i;
378 for (i=0; i<Length && i<StrBufSize-1; i++) {
379 vb = GetByte(Index);
380 if (! nullReached) {
381 if (vb == 0x00 || vb == nulChar ) {
382 nullReached = true; // either null or '@' (AIS null character)
383 StrBuf[i] = '\0';
384 } else {
385 StrBuf[i] = vb;
386 }
387 } else {
388 StrBuf[i] = '\0';
389 }
390 }
391 StrBuf[i] = '\0';
392 for (;i<Length;i++) GetByte(Index); // Stopped by buffer size, so read out bytes from message
393 for (;i<StrBufSize;i++) StrBuf[i] = '\0'; // Stopped by length, fill buffer with 0
394 return true;
395 } else return false;
396}

398//*****************************************************************************
399bool tN2kMsg::GetVarStr(size_t &StrBufSize, char *StrBuf, int &Index) const {
400 size_t Len=GetByte(Index)-2;
401 uint8_t Type=GetByte(Index);
402 if ( Type!=0x01 ) { StrBufSize=0; return false; }
403 if ( StrBuf!=0 ) {
404 GetStr(StrBufSize,StrBuf,Len,0xff,Index);

406 Index+=Len; // Just pass this string

449template<typename T>
450T byteswap(T val);
451
452template<>
453uint8_t byteswap(uint8_t val) {
454 return val;
455}
456
457template<>
458int8_t byteswap(int8_t val) {
459 return val;
460}
461
462template<>
463uint16_t byteswap(uint16_t val) {
464 return (val << 8) | (val >> 8);
465}
466
467template<>
468int16_t byteswap(int16_t val)
469{
470 return byteswap((uint16_t) val);
471}
472
473template<>
474uint32_t byteswap(uint32_t val) {
475 return ((val << 24)) |
476 ((val << 8) & 0xff0000UL) |
477 ((val >> 8) & 0xff00UL) |
478 ((val >> 24));
479}
480
481template<>
482int32_t byteswap(int32_t val) {
483 return byteswap((uint32_t) val);
484}
485
486template<>
487uint64_t byteswap(uint64_t val) {
488 return ((val << 56)) |
489 ((val << 40) & 0xff000000000000ULL) |
490 ((val << 24) & 0xff0000000000ULL) |
491 ((val << 8) & 0xff00000000ULL) |
492 ((val >> 8) & 0xff000000ULL) |
493 ((val >> 24) & 0xff0000ULL) |
494 ((val >> 40) & 0xff00ULL) |
495 ((val >> 56));
496}
497
498template<>
499int64_t byteswap(int64_t val) {
500 return byteswap((uint64_t) val);
501}

503//*****************************************************************************
504template<typename T>
505T GetBuf(size_t len, int& index, const unsigned char* buf) {
506 T v{0};
507
508 // This could be improved by casting the buffer to a pointer of T and
509 // doing a direct copy. That is, if unaligned data access is allowed.
510 memcpy(&v, &buf[index], len);
511 index += len;
512
513#if defined(HOST_IS_BIG_ENDIAN)
514 v = byteswap(v);
515#endif
516
517 return v;
518}
519
520//*****************************************************************************
521template<typename T>
522void SetBuf(T v, size_t len, int& index, unsigned char* buf) {
523#if defined(HOST_IS_BIG_ENDIAN)
524 v = byteswap(v);
525#endif
526
527 // This could be improved by casting the buffer to a pointer of T and
528 // doing a direct copy. That is, if unaligned data access is allowed.
529 memcpy(&buf[index], &v, len);
530 index += len;
531}

568//*****************************************************************************
569void SetBuf8ByteDouble(double v, double precision, int &index, unsigned char *buf) {
570 int64_t vll;
571 if ( !N2kIsNA(v) ) {
572 if ( sizeof(double)<8 ) {
573 double fp=precision*1e6;
574 int64_t fpll=1/fp;
575 vll=v*1e6L;
576 vll*=fpll;
577 } else {
578 vll=v/precision;
579 }
580 } else {
581 vll=N2kInt64NA;
582 }
583 SetBuf(vll, 8, index, buf);
584}

833//*****************************************************************************
834// Actisense Format:
835// <10><02><93><length (1)><priority (1)><PGN (3)><destination (1)><source (1)><time (4)><len (1)><data (len)><CRC (1)><10><03>
836void tN2kMsg::SendInActisenseFormat(N2kStream *port) const {
837 unsigned long _PGN=PGN;
838 unsigned long _MsgTime=MsgTime;
839 uint8_t msgIdx=0;
840 int byteSum = 0;
841 uint8_t CheckSum;
842 unsigned char ActisenseMsgBuf[MaxActisenseMsgBuf];
843
844 if (port==0 || !IsValid()) return;
845 // Serial.print("freeMemory()="); Serial.println(freeMemory());
846
847 ActisenseMsgBuf[msgIdx++]=Escape;
848 ActisenseMsgBuf[msgIdx++]=StartOfText;
849 AddByteEscapedToBuf(MsgTypeN2k,msgIdx,ActisenseMsgBuf,byteSum);
850 AddByteEscapedToBuf(DataLen+11,msgIdx,ActisenseMsgBuf,byteSum); //length does not include escaped chars
851 AddByteEscapedToBuf(Priority,msgIdx,ActisenseMsgBuf,byteSum);
852 AddByteEscapedToBuf(_PGN & 0xff,msgIdx,ActisenseMsgBuf,byteSum); _PGN>>=8;
853 AddByteEscapedToBuf(_PGN & 0xff,msgIdx,ActisenseMsgBuf,byteSum); _PGN>>=8;
854 AddByteEscapedToBuf(_PGN & 0xff,msgIdx,ActisenseMsgBuf,byteSum);
855 AddByteEscapedToBuf(Destination,msgIdx,ActisenseMsgBuf,byteSum);
856 AddByteEscapedToBuf(Source,msgIdx,ActisenseMsgBuf,byteSum);
857 // Time?
858 AddByteEscapedToBuf(_MsgTime & 0xff,msgIdx,ActisenseMsgBuf,byteSum); _MsgTime>>=8;
859 AddByteEscapedToBuf(_MsgTime & 0xff,msgIdx,ActisenseMsgBuf,byteSum); _MsgTime>>=8;
860 AddByteEscapedToBuf(_MsgTime & 0xff,msgIdx,ActisenseMsgBuf,byteSum); _MsgTime>>=8;
861 AddByteEscapedToBuf(_MsgTime & 0xff,msgIdx,ActisenseMsgBuf,byteSum);
862 AddByteEscapedToBuf(DataLen,msgIdx,ActisenseMsgBuf,byteSum);
863
864
865 for (int i = 0; i < DataLen; i++) AddByteEscapedToBuf(Data[i],msgIdx,ActisenseMsgBuf,byteSum);
866 byteSum %= 256;
867
868 CheckSum = (uint8_t)((byteSum == 0) ? 0 : (256 - byteSum));
869 ActisenseMsgBuf[msgIdx++]=CheckSum;
870 if (CheckSum==Escape) ActisenseMsgBuf[msgIdx++]=CheckSum;
871
872 ActisenseMsgBuf[msgIdx++] = Escape;
873 ActisenseMsgBuf[msgIdx++] = EndOfText;
874
875// if ( port->availableForWrite()>msgIdx ) { // 16.7.2017 did not work yet
876 port->write(ActisenseMsgBuf,msgIdx);
877// }
878 //Serial.print("Actisense data:");

880 //Serial.print("\r\n");
881}
