51 uint64_t N2kMillis64() {
52 static uint32_t RollCount=0;
53 static uint32_t LastRead=0;
54 uint32_t Now=millis();
55 if ( LastRead>Now ) RollCount++;
56 LastRead=Now;
57 return ((uint64_t)RollCount)<<32 | Now;
