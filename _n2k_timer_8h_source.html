238 /************************************************************************/
242 void UpdateNextTime() {
243 if ( Period==0 ) {
244 Disable();
245 return;
246 }
247 uint64_t now=N2kMillis64();
248 if ( (Offset+SyncOffset)>now ) {
249 NextTime=(Offset+SyncOffset);
250 } else {
251 uint64_t n=(now-(Offset+SyncOffset))/Period;
252 NextTime=SyncOffset+Offset+(n+1)*Period;
253 }
254 }
