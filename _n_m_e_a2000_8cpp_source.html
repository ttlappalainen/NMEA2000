716//*****************************************************************************
717void tNMEA2000::SetConfigurationInformation(const char *ManufacturerInformation,
718 const char *InstallationDescription1,
719 const char *InstallationDescription2) {
720 if ( LocalConfigurationInformationData!=0 ) free(LocalConfigurationInformationData); // This happens on second call, which is not good.
721 LocalConfigurationInformationData=0;
722
723 size_t ManInfoLen=(ManufacturerInformation?strlen(ManufacturerInformation)+1:0);
724#if !defined(N2K_NO_GROUP_FUNCTION_SUPPORT)
725 size_t InstDesc1Len=Max_N2kConfigurationInfoField_len;
726 size_t InstDesc2Len=Max_N2kConfigurationInfoField_len;
727#else
728 size_t InstDesc1Len=(InstallationDescription1?strlen(InstallationDescription1)+1:0);
729 size_t InstDesc2Len=(InstallationDescription2?strlen(InstallationDescription2)+1:0);
730#endif
731
732 if ( ManInfoLen>Max_N2kConfigurationInfoField_len ) ManInfoLen=Max_N2kConfigurationInfoField_len;
733 if ( InstDesc1Len>Max_N2kConfigurationInfoField_len ) InstDesc1Len=Max_N2kConfigurationInfoField_len;
734 if ( InstDesc2Len>Max_N2kConfigurationInfoField_len ) InstDesc2Len=Max_N2kConfigurationInfoField_len;
735
736 size_t TotalSize=ManInfoLen+InstDesc1Len+InstDesc2Len;
737 void *mem=(TotalSize>0?malloc(TotalSize):0);
738
739 LocalConfigurationInformationData=(char*)mem;
740 char *Info=LocalConfigurationInformationData;
741
742 SetCharBuf(InstallationDescription1,InstDesc1Len,Info);
743 ConfigurationInformation.InstallationDescription1=(InstallationDescription1?Info:0);
744 Info+=InstDesc1Len;
745
746 SetCharBuf(InstallationDescription2,InstDesc2Len,Info);
747 ConfigurationInformation.InstallationDescription2=(InstallationDescription2?Info:0);
748 Info+=InstDesc2Len;
749
750 SetCharBuf(ManufacturerInformation,ManInfoLen,Info);
751 ConfigurationInformation.ManufacturerInformation=(ManufacturerInformation?Info:0);
752}

784int tNMEA2000::GetSequenceCounter(unsigned long PGN, int iDev) {
785 if ( !IsValidDevice(iDev) ) return 0;
786
787 if ( Devices[iDev].PGNSequenceCounters==0 ) { // Sequence counters has not yet been initialized
788 Devices[iDev].MaxPGNSequenceCounters=GetFastPacketTxPGNCount(iDev)+1; // Reserve 1 for undefined PGNs
789 Devices[iDev].PGNSequenceCounters=new unsigned long[Devices[iDev].MaxPGNSequenceCounters];
790 for ( size_t i=0; i<Devices[iDev].MaxPGNSequenceCounters; i++ ) Devices[iDev].PGNSequenceCounters[i]=0;
791 }
792 if ( Devices[iDev].PGNSequenceCounters==0 ) return 0; // Should not be. Only in case of memory allocation problem.
793 size_t last=Devices[iDev].MaxPGNSequenceCounters-1;
794 unsigned long sc;
795 for ( size_t i=0; i<last; i++ ) {
796 if ( Devices[iDev].PGNSequenceCounters[i]==0 ) { // Empty place, use this
797 Devices[iDev].PGNSequenceCounters[i]=PGN;
798 return 0; // Start from sequence 0
799 }
800 if ( (Devices[iDev].PGNSequenceCounters[i]&0x00ffffff) == PGN ) { // Found counter, use it
801 sc=Devices[iDev].PGNSequenceCounters[i]>>24;
802 sc++; if (sc>7) sc=0; // Get next counter
803 Devices[iDev].PGNSequenceCounters[i]=PGN | (sc << 24);
804 return sc;
805 }
806 }
807 // PGN counter not found, so use common
808 sc=Devices[iDev].PGNSequenceCounters[last];
809 sc++; if (sc>7) sc=0;
810 Devices[iDev].PGNSequenceCounters[last]=sc;
811 return sc;
812}

1190 }

1192 }

1227void CanIdToN2k(unsigned long id, unsigned char &prio, unsigned long &pgn, unsigned char &src, unsigned char &dst) {
1228 unsigned char CanIdPF = (unsigned char) (id >> 16);
1229 unsigned char CanIdPS = (unsigned char) (id >> 8);
1230 unsigned char CanIdDP = (unsigned char) (id >> 24) & 1;
1231
1232 src = (unsigned char) id >> 0;
1233 prio = (unsigned char) ((id >> 26) & 0x7);
1234
1235 if (CanIdPF < 240) {
1236 /* PDU1 format, the PS contains the destination address */
1237 dst = CanIdPS;
1238 pgn = (((unsigned long)CanIdDP) << 16) | (((unsigned long)CanIdPF) << 8);
1239 } else {
1240 /* PDU2 format, the destination is implied global and the PGN is extended */
1241 dst = 0xff;
1242 pgn = (((unsigned long)CanIdDP) << 16) | (((unsigned long)CanIdPF) << 8) | (unsigned long)CanIdPS;
1243 }
1244}

1256unsigned long N2ktoCanID(unsigned char priority, unsigned long PGN, unsigned long Source, unsigned char Destination) {
1257 unsigned char CanIdPF = (unsigned char) (PGN >> 8);
1258
1259 if (CanIdPF < 240) { // PDU1 format
1260 if ( (PGN & 0xff) != 0 ) return 0; // for PDU1 format PGN lowest byte has to be 0 for the destination.
1261 return ( ((unsigned long)(priority & 0x7))<<26 | PGN<<8 | ((unsigned long)Destination)<<8 | (unsigned long)Source);
1262 } else { // PDU2 format
1263 return ( ((unsigned long)(priority & 0x7))<<26 | PGN<<8 | (unsigned long)Source);
1264 }

1400bool tNMEA2000::SendMsg(const tN2kMsg &N2kMsg, int DeviceIndex) {
1401 if ( dbMode==dm_None ) {
1402 if ( OpenState!=os_Open ) {
1403 if ( !(Open() && OpenState==os_Open) ) return false; // Can not do much
1404 }
1405 }
1406
1407 bool result=false;

1409 if ( DeviceIndex>=DeviceCount) return result;
1410 N2kMsg.CheckDestination();
1411 if (DeviceIndex>=0) { N2kMsg.ForceSource(Devices[DeviceIndex].N2kSource); } else { DeviceIndex=0; }
1412
1413 if ( N2kMsg.Source>N2kMaxCanBusAddress && N2kMsg.PGN!=N2kPGNIsoAddressClaim ) return false; // CAN bus address range is 0-251. Anyway allow ISO address claim mgs.
1414
1415 unsigned long canId=N2ktoCanID(N2kMsg.Priority,N2kMsg.PGN,N2kMsg.Source, N2kMsg.Destination);

1418// if (ForwardStream!=0 && ForwardType==tNMEA2000::fwdt_Text) { ForwardStream->print(F("Invalid PGN ")); ForwardStream->println(N2kMsg.PGN); }
1419 return false;
1420 }

1423
1424 if (N2kMsg.PGN==0) return false;
1425

1427 case dm_None:
1428 N2kMsgDbgStart("Send PGN:"); N2kMsgDbgln(N2kMsg.PGN);
1429 N2kMsgDbgStart(" - can ID:"); N2kMsgDbgln(canId);

1442#endif

1450 temp[0] = i|Order; //frame counter

1452 temp[1] = N2kMsg.DataLen; //total bytes in fast packet
1453 //send the first 6 bytes
1454 for (int j = 2; j<8; j++) {
1455 temp[j]=N2kMsg.Data[cur];
1456 cur++;
1457 }
1458 N2kPrintFreeMemory("SendMsg, fastpacket");

1460 int j=1;
1461 //send the next 7 data bytes
1462 for (; j<8 && cur<N2kMsg.DataLen; j++) {
1463 temp[j]=N2kMsg.Data[cur];
1464 cur++;
1465 }
1466 for (; j<8; j++) {
1467 temp[j]=0xff;
1468 }
1469 }
1470
1471 DbgPrintBuf(8,temp,true);
1472 result=SendFrame(canId, 8, temp, true);

1475 ForwardStream->print(F(", frame:")); ForwardStream->print(i); ForwardStream->print(F("/")); ForwardStream->print(frames);
1476 ForwardStream->println(F(" send failed"));
1477 }
1478 }
1479 }

1481 if ( ForwardOwnMessages() ) ForwardMessage(N2kMsg);
1482 break;
1483 case dm_ClearText:
1484 result=true;
1485 N2kMsg.Print(ForwardStream);
1486 break;
1487 case dm_Actisense:
1488 result=true;
1489 N2kMsg.SendInActisenseFormat(ForwardStream);
1490 break;
1491 }
1492
1493 return result;

1735//*****************************************************************************
1736bool tNMEA2000::TestHandleTPMessage(unsigned long PGN, unsigned char Source, unsigned char Destination,
1737 unsigned char len, unsigned char *buf,
1738 uint8_t &MsgIndex) {
1739 MsgIndex=MaxN2kCANMsgs;
1740 int iDev=FindSourceDeviceIndex(Destination);
1741
1742 if ( PGN==TP_CM ) {
1743 unsigned char TP_CM_Control=buf[0];
1744 int Index=5;
1745 unsigned long TransportPGN=GetBuf3ByteUInt(Index,buf);
1746
1747 switch (TP_CM_Control) {
1748 case TP_CM_BAM:
1749 case TP_CM_RTS: {
1750 N2kMsgDbgln("Got TP Command");
1751 Index=1;
1752 uint16_t nBytes=GetBuf2ByteUInt(Index,buf);
1753 uint8_t TPMaxPackets=buf[Index++];
1754 //Index++; // reserved
1755
1756 FindFreeCANMsgIndex(TransportPGN,Source,Destination,true,MsgIndex);
1757
1758 if (MsgIndex==MaxN2kCANMsgs) { // No free msg place
1759 N2kMsgDbgStart("No free msg slot"); N2kMsgDbgln();
1760 if ( (TP_CM_Control==TP_CM_RTS) && (iDev>=0) ) { // If it was for us and not broadcast, we need to abort transport
1761 SendTPCM_Abort(TransportPGN,Source,iDev,TP_CM_AbortBusy);
1762 }
1763 } else { // Start transport
1764 N2kMsgDbgStart("Use msg slot: "); N2kMsgDbgln(MsgIndex);
1765 bool FastPacket;
1766 N2kCANMsgBuf[MsgIndex].KnownMessage=CheckKnownMessage(TransportPGN,N2kCANMsgBuf[MsgIndex].SystemMessage,FastPacket);
1767 if ( nBytes < tN2kMsg::MaxDataLen && // Currently we can handle only tN2kMsg::MaxDataLen long messages
1768 (N2kCANMsgBuf[MsgIndex].KnownMessage || !HandleOnlyKnownMessages()) ) {
1769 N2kCANMsgBuf[MsgIndex].FreeMsg=false;
1770 N2kCANMsgBuf[MsgIndex].N2kMsg.Init(7 /* Priority? */,TransportPGN,Source,Destination);
1771 N2kCANMsgBuf[MsgIndex].CopiedLen=0;
1772 N2kCANMsgBuf[MsgIndex].LastFrame=0;
1773 N2kCANMsgBuf[MsgIndex].N2kMsg.DataLen=nBytes;
1774 N2kCANMsgBuf[MsgIndex].N2kMsg.SetIsTPMessage();
1775 N2kCANMsgBuf[MsgIndex].TPMaxPackets=TPMaxPackets;
1776 if ( (TP_CM_Control==TP_CM_RTS) && (iDev>=0) ) { // If it was for us and not broadcast, we need to response
1777 SendTPCM_CTS(TransportPGN,Source,iDev,N2kCANMsgBuf[MsgIndex].TPMaxPackets,N2kCANMsgBuf[MsgIndex].LastFrame+1);
1778 N2kCANMsgBuf[MsgIndex].TPRequireCTS=TPCtsPackets(N2kCANMsgBuf[MsgIndex].TPMaxPackets);
1779 } else {
1780 N2kCANMsgBuf[MsgIndex].TPMaxPackets=0xff; // TPMaxPackets>0 indicates that it is TP message
1781 }
1782 } else { // Too long or unknown
1783 if ( (TP_CM_Control==TP_CM_RTS) && (iDev>=0) ) { // If it was for us and not broadcast, we need to response
1784 SendTPCM_Abort(TransportPGN,Source,iDev,TP_CM_AbortBusy); // Abort
1785 }
1786 }
1787 }
1788 break;
1789 }
1790 case TP_CM_CTS:
1791 if ( !IsValidDevice(iDev) ) break; // Should never fail
1792 N2kMsgDbgStart("Got TP CTS"); N2kMsgDbgln(MsgIndex);
1793 if ( IsBroadcast(Devices[iDev].PendingTPMsg.Destination) ) break; // We should not get controls for broadcast TP msg
1794 if ( Devices[iDev].PendingTPMsg.PGN!=TransportPGN ) { // Some failure on communication
1795 EndSendTPMessage(iDev); // Should we retry from beginning?
1796 break;
1797 }
1798 // Now respond with next data packets
1799 if ( buf[1]>0 ) { // Note that with 0, receiver wants to have break
1800 if ( buf[2]-1!=Devices[iDev].NextDTSequence ) { // We got sequence error
1801 EndSendTPMessage(iDev); // Should we retry from beginning?
1802 break;
1803 }
1804 uint8_t MaxTPSequences=buf[1];
1805 bool TPDTResult=true;
1806 for ( uint8_t iSeq=0; TPDTResult &&iSeq<MaxTPSequences && !HasAllTPDTSent(iDev); iSeq++ ) TPDTResult&=SendTPDT(iDev);
1807 if ( !TPDTResult ) EndSendTPMessage(iDev);
1808 }
1809 Devices[iDev].NextDTSendTime.FromNow(100); // Set timeout for next response
1810 break;
1811 case TP_CM_ACK:
1812 if ( !IsValidDevice(iDev) ) break; // Should never fail
1813 N2kMsgDbgStart("Got TP ACK"); N2kMsgDbgln(MsgIndex);
1814 if ( IsBroadcast(Devices[iDev].PendingTPMsg.Destination) ) break; // We should not get controls for broadcast TP msg
1815 EndSendTPMessage(iDev);
1816 break;
1817 case TP_CM_Abort:
1818 if ( !IsValidDevice(iDev) ) break; // Should never fail
1819 N2kMsgDbgStart("Got TP Abort"); N2kMsgDbgln(MsgIndex);
1820 if ( IsBroadcast(Devices[iDev].PendingTPMsg.Destination) ) break; // We should not get controls for broadcast TP msg
1821 EndSendTPMessage(iDev);
1822 break;
1823 default:
1824 ;
1825 }
1826 MsgIndex=MaxN2kCANMsgs; // After TP_CM, message is newer ready.
1827 return true;
1828 } else if ( PGN==TP_DT ) { // Datapacket
1829 N2kMsgDbgStart("Got TP data"); N2kMsgDbgln(MsgIndex);
1830 // So we need to find TP msg which sender and destination matches.
1831 for (MsgIndex=0;
1832 MsgIndex<MaxN2kCANMsgs &&
1833 !( !N2kCANMsgBuf[MsgIndex].FreeMsg
1834 && N2kCANMsgBuf[MsgIndex].N2kMsg.IsTPMessage()
1835 && N2kCANMsgBuf[MsgIndex].N2kMsg.Destination==Destination
1836 && N2kCANMsgBuf[MsgIndex].N2kMsg.Source==Source
1837 );
1838 MsgIndex++);
1839 // if (MsgIndex==MaxN2kCANMsgs) N2kMsgDbgln("TP data msg not found");
1840 // for (int i=1; i<len; i++) N2kMsgDbgln(buf[i]);
1841 if (MsgIndex<MaxN2kCANMsgs) { // found TP message under reception
1842 N2kMsgDbgStart("Use msg slot: "); N2kMsgDbgln(MsgIndex);
1843 if (N2kCANMsgBuf[MsgIndex].LastFrame+1 == buf[0]) { // Right packet is coming
1844 // Add packet to the message
1845 CopyBufToCANMsg(N2kCANMsgBuf[MsgIndex],1,len,buf);
1846 N2kCANMsgBuf[MsgIndex].LastFrame=buf[0];
1847 // Transport protocol is slower, so to avoid timeout, we reset message time
1848 N2kCANMsgBuf[MsgIndex].N2kMsg.MsgTime=N2kMillis();
1849 if ( N2kCANMsgBuf[MsgIndex].CopiedLen>=N2kCANMsgBuf[MsgIndex].N2kMsg.DataLen ) { // all done
1850 N2kCANMsgBuf[MsgIndex].Ready=true;
1851 if ( N2kCANMsgBuf[MsgIndex].TPRequireCTS>0 && iDev>=0 ) { // send response
1852 SendTPCM_EndAck(N2kCANMsgBuf[MsgIndex].N2kMsg.PGN,Source,iDev,N2kCANMsgBuf[MsgIndex].N2kMsg.DataLen,N2kCANMsgBuf[MsgIndex].LastFrame);
1853 }
1854 } else {
1855 if ( N2kCANMsgBuf[MsgIndex].TPRequireCTS>0 && ((N2kCANMsgBuf[MsgIndex].LastFrame)%N2kCANMsgBuf[MsgIndex].TPRequireCTS)==0 ) { // send response
1856 SendTPCM_CTS(N2kCANMsgBuf[MsgIndex].N2kMsg.PGN,Source,iDev,N2kCANMsgBuf[MsgIndex].TPMaxPackets,N2kCANMsgBuf[MsgIndex].LastFrame+1);
1857 }
1858 }
1859 } else { // Wrong packet - either we lost packet or sender sends wrong, so free this
1860 N2kMsgDbgStart("Invalid packet: "); N2kMsgDbgln(buf[0]);
1861 if ( N2kCANMsgBuf[MsgIndex].TPRequireCTS>0 && iDev>=0 ) { // We need to abort transport
1862 SendTPCM_Abort(N2kCANMsgBuf[MsgIndex].N2kMsg.PGN,Source,iDev,TP_CM_AbortTimeout); // Abort transport
1863 }
1864 N2kCANMsgBuf[MsgIndex].FreeMessage();
1865
1866 }
1867 if ( !N2kCANMsgBuf[MsgIndex].Ready ) MsgIndex=MaxN2kCANMsgs;
1868 }
1869 return true; // We handled message
1870 }
1871
1872 return false;
1873}

1875//*****************************************************************************
1876bool tNMEA2000::StartSendTPMessage(const tN2kMsg& msg, int iDev) {
1877 if ( !IsValidDevice(iDev) ) return false;
1878
1879 if ( Devices[iDev].PendingTPMsg.PGN!=0 ) return false; // No room for sending TP message
1880
1881 int result=false;
1882
