27// Define schedulers for messages. Define schedulers here disabled. Schedulers will be enabled
28// on OnN2kOpen so they will be synchronized with system.
29// We use own scheduler for each message so that each can have different offset and period.
30// Setup periods according PGN definition (see comments on IsDefaultSingleFrameMessage and
31// IsDefaultFastPacketMessage) and message first start offsets. Use a bit different offset for
32// each message so they will not be sent at same time.
33tN2kSyncScheduler TemperatureScheduler(false,2000,500);
34tN2kSyncScheduler EnvironmentalScheduler(false,500,510);
35tN2kSyncScheduler OutsideEnvironmentalScheduler(false,500,520);

42 TemperatureScheduler.UpdateNextTime();

97void SendN2kTemperature() {
98 tN2kMsg N2kMsg;
99
100 if ( TemperatureScheduler.IsTime() ) {
101 TemperatureScheduler.UpdateNextTime();
102 SetN2kTemperature(N2kMsg, 1, 1, N2kts_MainCabinTemperature, ReadCabinTemp());
103 NMEA2000.SendMsg(N2kMsg);
104 }
105
106 if ( EnvironmentalScheduler.IsTime() ) {
107 EnvironmentalScheduler.UpdateNextTime();
108 SetN2kEnvironmentalParameters(N2kMsg, 1, N2kts_MainCabinTemperature, ReadCabinTemp());
109 NMEA2000.SendMsg(N2kMsg);
110 }
111
112 if ( OutsideEnvironmentalScheduler.IsTime() ) {
113 OutsideEnvironmentalScheduler.UpdateNextTime();
114 SetN2kOutsideEnvironmentalParameters(N2kMsg, 1, ReadWaterTemp());
115 NMEA2000.SendMsg(N2kMsg);
116 // Serial.print(millis()); Serial.println(", Temperature send ready");
117 }
118}
